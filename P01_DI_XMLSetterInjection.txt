<beans xmlns="http://www.springframework.org/schema/beans"
       xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"

        http://www.springframework.org/schema/beans http://www.springframework.org/schema/beans/spring-beans.xsd">

    <bean id="GreetingServiceObject" class="com.codeX.GreetingService">
           <property name="message" ref="MessageObject"/>
    </bean>

    <bean id="MessageObject" class="com.codeX.Message">
        <property name="messageText" value="Hello World, I am Spring Developer" />
    </bean>

</beans>




package com.codeX;

public class Message {

    private String messageText ;

    public String getMessageText() {
        return messageText;
    }

    public void setMessageText(String messageText) {
        this.messageText = messageText;
    }
}



package com.codeX;

public class GreetingService {

    private Message message;

    public Message getMessage() {
        return message;
    }

    public void setMessage(Message message) {
        this.message = message;
    }

    
}



package com.codeX;

import org.springframework.context.ApplicationContext;
import org.springframework.context.support.ClassPathXmlApplicationContext;

public class App 
{
    public static void main( String[] args ) {
        ApplicationContext context = new ClassPathXmlApplicationContext("springConfig.xml");

        //Demo Dependency Injection
        GreetingService greetingService = (GreetingService)context.getBean("GreetingServiceObject");
        Message message = greetingService.getMessage();
        System.out.println(message.getMessageText());
        message.setMessageText("Hi I am Java Developer, I build apps with Spring and Flutter");
        System.out.println(message.getMessageText());




    }
}
