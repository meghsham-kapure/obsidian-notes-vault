package com.codex.JournalApp.model;


import org.springframework.data.annotation.Id;
import org.springframework.data.mongodb.core.mapping.Document;
import org.springframework.stereotype.Component;

import java.util.Objects;

@Document(collation = "Journals" )
public class JournalModel {
    @Id
    private String journalId;
    private String journalTitle;
    private String journalText;
    private String journalDate;

    public JournalModel() {

    }

    public JournalModel(String journalId, String journalTitle, String journalText, String journalDate) {
        this.journalId = journalId;
        this.journalTitle = journalTitle;
        this.journalText = journalText;
        this.journalDate = journalDate;
    }

    @Override
    public String toString() {
        return "JournalModel{" +
                "journalId='" + journalId + '\'' +
                ", journalTitle='" + journalTitle + '\'' +
                ", journalText='" + journalText + '\'' +
                ", journalDate='" + journalDate + '\'' +
                '}';
    }

    @Override
    public int hashCode() {
        return Objects.hashCode(journalId);
    }

    public String getJournalId() {
        return journalId;
    }

    public void setJournalId(String journalId) {
        this.journalId = journalId;
    }

    public String getJournalTitle() {
        return journalTitle;
    }

    public void setJournalTitle(String journalTitle) {
        this.journalTitle = journalTitle;
    }

    public String getJournalText() {
        return journalText;
    }

    public void setJournalText(String journalText) {
        this.journalText = journalText;
    }

    public String getJournalDate() {
        return journalDate;
    }

    public void setJournalDate(String journalDate) {
        this.journalDate = journalDate;
    }
}



package com.codex.JournalApp.controller;

import com.codex.JournalApp.model.JournalModel;
import com.codex.JournalApp.service.JournalService;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.web.bind.annotation.*;

@RestController
@RequestMapping("/journalApp")
public class JournalController {

    @Autowired
    JournalService journalService;

    @GetMapping("/healthCheck")
    public String healthCheck(){
        return "OK";
    }

    @PostMapping("/createJournal")
    public JournalModel createJournal (@RequestBody JournalModel journalEntry){
        return journalService.createJournal(journalEntry);
    }

    PutMapping("/updateJournal/{Id}")
    public String updateModel (@PathVariable String Id, @RequestBody JournalModel journalEntry){
        return JournalService.updateModel(Id, journalEntry);
    }
}
