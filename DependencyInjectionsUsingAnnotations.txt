package com.codex.dependencyinjectionsusingannotations;

import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;
import org.springframework.context.ApplicationContext;

@SpringBootApplication
public class DependencyInjectionsUsingAnnotationsApplication {

    public static void main(String[] args) {
        ApplicationContext context =
                SpringApplication.run(DependencyInjectionsUsingAnnotationsApplication.class, args);

         Car car = context.getBean(Car.class);
         car.showCar();
    }
}


package com.codex.dependencyinjectionsusingannotations;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

@Component
public class Car {
    @Autowired
    Wheel wheel;

    public void showCar(){
        wheel.showWheels();
    }
}

package com.codex.dependencyinjectionsusingannotations;

   void showWheels(){
       System.out.println("Showing Wheels");
   }
